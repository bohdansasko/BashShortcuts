# BashShortcuts
```bash
#!/bin/bash

# git
alias gaa='git add *'
alias gs='git status'
alias gb='git branch'
alias gp='git pull'
alias gl='git log'

alias gamend='git commit --amend'
alias gamendNoEdit='git commit --amend --no-edit'

# system
alias ll='ls -la'
alias h='history'
alias c='clear'
alias ..='cd ..'

alias homeDir='cd ~/'

alias editBash='open ~/.bash_profile'
alias reloadBash='source ~/.bash_profile'

# pods
alias podi='pod install'
alias podu='pod update'
```
